import java.util.Scanner;
public class Main{
	public static void main (String[] args) 
	{
	    Scanner scan = new Scanner(System.in);
	    int pass=scan.nextInt();
	    int fare=scan.nextInt();
	    int acpass=scan.nextInt();
	    int acfare=scan.nextInt();
	    int platform=scan.nextInt();
	    int maintenance=scan.nextInt();
	    int fuel=scan.nextInt();
	    int railway=(pass*fare)+(acpass*acfare)+platform-maintenance-fuel;
	    System.out.println(railway);
	    }
}
